def crcNmea(data):
    """Calculate the NMEA 0183 checksum (XOR of all data bytes).

    Parameters
    ----------
    data : bitstring.BitStream
        The bytes between ``$`` and ``*`` of an NMEA sentence. Must be a whole
        number of bytes.

    Returns
    -------
    bitstring.BitStream
        The 8-bit checksum, matching the two hex digits after ``*``.
    """
    crc = BitStream("0x00")
    length = int(data.length / 8)
    for _ in range(length):
        crc = crc ^ data.read("bytes:1")
    return crc

def genLookup():
    """
    Generate lookup table for Qualcomm 24 bit cyclic redundancy check using
    0x1864CFB as 1+24 bit polynomia.

        𝑝(𝑋) = 𝑋^23 + 𝑋^17 + 𝑋^13 + 𝑋^12 + 𝑋^11 + 𝑋^9 + 𝑋^8 + 𝑋^7 + 𝑋^5
                + 𝑋^3 + 1
        p = 0b 1000 0010 0011 1011 1010 1001
          = 0x823BA9

    Returns
    -------
    Array with 24 bit remainders

    """
    crcWidth = 24
    crcPoly = 0x864CFB
    topbit = 1 << (crcWidth - 1)
    lookup = []

    for index in range(256):
        r = index << (crcWidth - 8)
        for _ in range(8):
            if r & topbit:
                r = (r << 1) ^ crcPoly
            else:
                r <<= 1
        lookup.append(r & ((((1 << (crcWidth - 1)) - 1) << 1) | 1))
    return lookup
